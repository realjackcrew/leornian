# Docker Commands
docker compose down -v  # Stops and removes containers, networks, and volumes
docker compose up -d    # Starts containers in detached mode

# Database Commands
npx prisma generate     # Generates Prisma Client
npx prisma db push      # Pushes schema changes to database
npx prisma studio       # Opens Prisma Studio to view/edit database
npx prisma migrate dev  # Creates and applies migrations
npx prisma migrate reset # Resets database and applies migrations

# Server Commands
npm run dev            # Starts development server
npm run build          # Builds production server
npm start              # Starts production server
npm install            # Installs dependencies
npm install -D typescript @types/node @types/express # Installs TypeScript and types
